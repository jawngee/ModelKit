23 +(void)bindModel:(MKitModel *)model data:(NSDictionary *)data
24 {
25  MKitReflectedClass *ref=[MKitReflectionManager reflectionForClass:[model class] ignorePropPrefix:@"model" recurseChainUntil:[MKitModel class]];
26 
27  if ((data[@"modelId"]) && (data[@"modelId"]!=[NSNull null]))
28  model.modelId=data[@"modelId"];
29 
30  if ((data[@"createdAt"]) && (data[@"createdAt"]!=[NSNull null]))
31  model.createdAt=[NSDate dateFromISO8601:data[@"createdAt"]];
32 
33  if ((data[@"updatedAt"]) && (data[@"updatedAt"]!=[NSNull null]))
34  model.updatedAt=[NSDate dateFromISO8601:data[@"updatedAt"]];
35 
36  [ref.properties enumerateKeysAndObjectsUsingBlock:^(id key, id obj, BOOL *stop) {
37  MKitReflectedProperty *prop=obj;
38 
39  if (data[prop.name]==[NSNull null])
40  {
41  [model setValue:nil forKey:prop.name];
42  }
43  else if ([prop.typeClass isSubclassOfClass:[MKitModel class]])
44  {
45  MKitModel *m=nil;
46  NSDictionary *md=[data objectForKey:prop.name];
47  if ((md[@"__type"]) && ([md[@"__type"] isEqualToString:@"Object"]))
48  {
49  m=[prop.typeClass instanceWithObjectId:md[@"objectId"]];
50  [self bindModel:m data:md];
51  }
52  else if ((md[@"__type"]) && ([md[@"__type"] isEqualToString:@"Pointer"]))
53  {
54  m=[prop.typeClass instanceWithObjectId:md[@"objectId"]];
55  if ((m.modelState==ModelStateNeedsData) && ([prop.typeClass isSubclassOfClass:[MKitServiceModel class]]))
56  [((MKitServiceModel *)m) fetchInBackground:nil];
57  }
58 
59  [model setValue:m forKey:prop.name];
60  }
61  else if ([prop.typeClass isSubclassOfClass:[MKitMutableModelArray class]])
62  {
63  NSArray *vals=data[prop.name];
64  if (vals)
65  {
66  MKitMutableModelArray *objs=[MKitMutableModelArray array];
67 
68  for(NSDictionary *md in vals)
69  {
70  NSString *cname=md[@"className"];
71  Class c=[MKitModelRegistry registeredClassForModel:cname];
72  if (!c)
73  c=NSClassFromString(cname);
74  if (!c)
75  @throw [NSException exceptionWithName:@"Invalid Class Name" reason:[NSString stringWithFormat:@"Parse returned an unknown classname '%@'.",cname] userInfo:md];
76 
77  MKitModel *m=nil;
78  m=[c instanceWithObjectId:md[@"objectId"]];
79  if ((md[@"__type"]) && ([md[@"__type"] isEqualToString:@"Object"]))
80  [self bindModel:m data:md];
81  else if ((md[@"__type"]) && ([md[@"__type"] isEqualToString:@"Pointer"]))
82  {
83  if ((m.modelState==ModelStateNeedsData) && ([prop.typeClass isSubclassOfClass:[MKitServiceModel class]]))
84  [((MKitServiceModel *)m) fetchInBackground:nil];
85  }
86 
87  [objs addObject:m];
88  }
89 
90  [model setValue:objs forKey:prop.name];
91  }
92  else
93  [model setValue:nil forKey:prop.name];
94  }
95  else if ([prop.typeClass isSubclassOfClass:[NSDate class]])
96  {
97  [model setValue:[NSDate dateFromISO8601:data[prop.name]] forKey:prop.name];
98  }
99  else
100  {
101  [model setValue:data[prop.name] forKey:prop.name];
102  }
103  }];
104 
105  model.modelState=ModelStateValid;
106  [model resetChanges];
107 }

109 +(NSArray *)bindArrayOfModels:(NSArray *)models forClass:(Class)modelClass
110 {
111  NSMutableArray *result=[NSMutableArray array];
112  for(NSDictionary *d in models)
113  {
114  MKitModel *m=[modelClass instanceWithObjectId:d[@"objectId"]];
115 
116  [self bindModel:m data:d];
117 
118  [result addObject:m];
119  }
120 
121  return result;
122 }
