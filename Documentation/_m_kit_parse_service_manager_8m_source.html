151 {
152  AFHTTPRequestOperation *op=[self modelRequestWithMethod:@"PUT" model:model params:nil body:[props JSONData]];
153 
154  [op start];

156 
157  if ([op hasAcceptableStatusCode])
158  {
159  NSDictionary *data=[op.responseString objectFromJSONString];

168  *error=op.error;

174 {
175  AFHTTPRequestOperation *op=[self classRequestWithMethod:@"POST" class:[model class] params:nil body:[props JSONData]];
176

179

184  model.objectId=[data objectForKey:@"objectId"];
185  model.createdAt=created;
186  model.updatedAt=created;
187  model.modelState=ModelStateValid;
188

191  {
192  ((MKitParseUser *)model).modelSessionToken=data[@"sessionToken"];
193 
194  // save the credentials in the keychain
195  [self storeUserCredentials:(MKitParseUser *)model];
196  }
197 
198  return YES;
199  }

201  // TODO: Handle JSON error response

203  *error=op.error;

208 -(BOOL)saveModel:(MKitModel *)model error:(NSError **)error
209 {
210  // normally I don't comment this much, but since you are reading this
211  // you are possibly trying to write a new backend for ModelKit. Hopefully
212  // these comments will be helpful to you
213 
214  // If saving new, we want all of the properties, otherwise just the ones that have changed
215  NSDictionary *props=(model.modelState==ModelStateNew) ? [model properties] : model.modelChanges;
216 
217  NSMutableDictionary *refs=[NSMutableDictionary dictionary];
218  NSMutableDictionary *propsToSave=[NSMutableDictionary dictionary];
219 
220  // Not sure we need to do this, but better safe than sorry. Will revisit.
221  [propsToSave setObject:model.modelId forKey:@"modelId"];
222 
223  // Loop through all of the properties we are saving
224  [props enumerateKeysAndObjectsUsingBlock:^(id key, id obj, BOOL *stop) {
225 
226  if ([[obj class] isSubclassOfClass:[MKitModel class]])
227  {
228  MKitModel *m=(MKitModel *)obj;
229 
230  // If the model has an object id then it has been saved before
231  if (m.objectId)
232  {
233  if (m.modelState==ModelStateDirty)
234  {
235  // we need to save this
236  [refs setObject:obj forKey:key];
237  }
238  else if (m.modelState!=ModelStateDeleted)
239  {
240  // no need to save, so we just turn it into the "pointer" format parse uses
241  [propsToSave setObject:[m parsePointer] forKey:key];
242  }
243  else
244  {
245  // set it to NULL
246  [propsToSave setObject:[NSNull null] forKey:key];
247  }
248  }
249  else
250  {
251  // model has never been saved
252  [refs setObject:obj forKey:key];
253  }
254  }
255  else if ([[obj class] isSubclassOfClass:[MKitMutableModelArray class]])
256  {
257  NSMutableArray *toSave=nil;
258 
259  // Convert what we can to parse "pointers", the rest we will process later
260  NSArray *pointerArray=[obj parsePointerArray:&toSave];
261 
262  // Save the ones we need to process later
263  if (toSave!=nil)
264  [refs setObject:toSave forKey:key];
265 
266  // We'll save the ones we can now
267  if (pointerArray.count>0)
268  [propsToSave setObject:pointerArray forKey:key];
269  }
270  else if ([[obj class] isSubclassOfClass:[NSDate class]])
271  {
272  // convert to parse's date format
273  [propsToSave setObject:@{@"__type":@"Date",@"iso":[((NSDate *)obj) ISO8601String]} forKey:key];
274  }
275  else if ([[obj class] isSubclassOfClass:[MKitParseFile class]])
276  {
277  [propsToSave setObject:[((MKitParseFile *)obj) parseFilePointer] forKey:key];
278  }
279  else if ([[obj class] isSubclassOfClass:[NSMutableArray array]])
280  {
281  NSMutableArray *arrayCopy=[NSMutableArray array];
282  for(id val in ((NSMutableArray *)obj))
283  if ([[val class] isSubclassOfClass:[NSDate class]])
284  [arrayCopy addObject:@{@"__type":@"Date",@"iso":[((NSDate *)val) ISO8601String]}];
285  else if ([[val class] isSubclassOfClass:[MKitParseFile class]])
286  [arrayCopy addObject:[((MKitParseFile *)val) parseFilePointer]];
287  else
288  [arrayCopy addObject:val];
289 
290  [propsToSave setObject:arrayCopy forKey:key];
291  }
292  else
293  [propsToSave setObject:obj forKey:key];
294  }];
295 
296  BOOL result=NO;
297 
298  if (!model.objectId)
299  {
300  // this model is "new" and needs to be saved
301  result=[self internalSaveModel:model props:propsToSave error:error];
302  }
303  else if (propsToSave.count>0)
304  {
305  // just needs to be updated
306  result=[self internalUpdateModel:model props:propsToSave error:error];
307  }
308 
309  // if we failed we're going to bail here.
310  if (!result)
311  return NO;
312 
313  [propsToSave removeAllObjects];
314 
315  // process all of the other models this model contains/points to
316  if (refs.count>0)
317  {
318  __block BOOL blockResult=YES;
319 
320  [refs enumerateKeysAndObjectsUsingBlock:^(id key, id obj, BOOL *stop) {
321  if ([[obj class] isSubclassOfClass:[MKitModel class]])
322  {
323  MKitModel *m=(MKitModel *)obj;
324  blockResult=[self saveModel:m error:error];
325  if (!blockResult)
326  return;
327 
328  [propsToSave setObject:[m parsePointer] forKey:key];
329  }
330  else if ([[obj class] isSubclassOfClass:[NSMutableArray class]])
331  {
332  MKitMutableModelArray *savedModels=[MKitMutableModelArray array];
333  for(MKitModel *m in ((NSMutableArray *)obj))
334  {
335  blockResult=[self saveModel:m error:error];
336  if (!blockResult)
337  return;
338 
339  [savedModels addObject:m];
340  }
341 
342  NSMutableArray *toSave=nil;
343  [propsToSave setObject:[savedModels parsePointerArray:&toSave] forKey:key];
344  }
345  }];
346 
347  if (!blockResult)
348  return NO;
349 
350  // now update this model with the changes
351  return [self internalUpdateModel:model props:propsToSave error:error];
352  }
353 
354  return YES;
355 }

377 -(BOOL)fetchModel:(MKitModel *)model error:(NSError **)error
378 {
379  MKitReflectedClass *ref=[MKitReflectionManager reflectionForClass:[model class] ignorePropPrefix:@"model" recurseChainUntil:[MKitModel class]];
380 
381  // We want Parse to return the full object data for all models this model points to/contains
382  NSMutableArray *toInclude=[NSMutableArray array];
383  for(MKitReflectedProperty *p in [ref.properties allValues])
384  {
385  if (([p.typeClass isSubclassOfClass:[MKitModel class]]) || ([p.typeClass isSubclassOfClass:[MKitMutableModelArray class]]))
386  [toInclude addObject:p.name];
387  }
388 
389  NSDictionary *params=nil;
390 
391  if (toInclude.count>0)
392  params=@{@"include":[toInclude componentsJoinedByString:@","]};
393 
394  AFHTTPRequestOperation *op=[self modelRequestWithMethod:@"GET" model:model params:params body:nil];
395 
396  [op start];
397  [op waitUntilFinished];
398 
399  if ([op hasAcceptableStatusCode])
400  {
401  id data=[op.responseString objectFromJSONString];
402  [MKitParseModelBinder bindModel:model data:data];
403  return YES;
404  }
405 
406  if (error!=nil)
407  *error=op.error;
408 
409  return NO;
410 }

434 -(BOOL)saveFile:(MKitServiceFile *)file progressBlock:(MKitProgressBlock)progressBlock error:(NSError **)error
435 {
436  if (file.state!=FileStateNew)
437  return YES;
438 
439  AFHTTPRequestOperation *op=[self requestWithMethod:@"POST"
440  path:[NSString stringWithFormat:@"files/%@",file.name]
441  params:nil
442  body:file.data
443  contentType:file.contentType];
444 
445  if (progressBlock)
446  [op setUploadProgressBlock:^(NSUInteger bytesWritten, long long totalBytesWritten, long long totalBytesExpectedToWrite) {
447  progressBlock((float)totalBytesWritten/(float)totalBytesExpectedToWrite);
448  }];
449 
450  [op start];
451  [op waitUntilFinished];
452 
453  if ([op hasAcceptableStatusCode])
454  {
455  NSDictionary *result=[op.responseString objectFromJSONString];
456  file.url=result[@"url"];
457  file.name=result[@"name"];
458  file.state=FileStateSaved;
459  file.data=nil;
460 
461  return YES;
462  }
463 
464  if (error!=nil)
465  *error=op.error;
466
